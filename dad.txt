    <script>
        let currentUser = null;
        let isDemoMode = false;
        let activeBalls = 0; // –°—á–µ—Ç—á–∏–∫ –∞–∫—Ç–∏–≤–Ω—ã—Ö —à–∞—Ä–∏–∫–æ–≤

        // –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è
        document.addEventListener('DOMContentLoaded', function() {
            initializeGame();
            createPlinkoField();
            
            // –ü–µ—Ä–µ—Å–æ–∑–¥–∞–µ–º –ø–æ–ª–µ –ø—Ä–∏ –∏–∑–º–µ–Ω–µ–Ω–∏–∏ —Ä–∞–∑–º–µ—Ä–∞ –æ–∫–Ω–∞
            window.addEventListener('resize', createPlinkoField);
        });

        function goBack() {
            window.location.href = 'index.html';
        }

        function initializeGame() {
            const tg = window.Telegram.WebApp;
            if (tg.initDataUnsafe && tg.initDataUnsafe.user) {
                currentUser = {
                    id: tg.initDataUnsafe.user.id,
                    username: tg.initDataUnsafe.user.username || `User_${tg.initDataUnsafe.user.id}`
                };
                loadUserData();
            }
        }

        async function loadUserData() {
            try {
                const response = await fetch(`/api/user/balance/${currentUser.id}`);
                if (response.ok) {
                    const userData = await response.json();
                    const balance = userData.demo_mode ? userData.demo_balance : userData.main_balance;
                    updateBalanceDisplay(balance);
                    isDemoMode = userData.demo_mode;
                    document.getElementById('demo-badge').style.display = isDemoMode ? 'block' : 'none';
                }
            } catch (error) {
                console.error('Error loading user data:', error);
            }
        }

        function updateBalanceDisplay(balance) {
            document.getElementById('balance').textContent = balance.toFixed(2);
            document.getElementById('current-balance').textContent = balance.toFixed(2);
        }

        // –°–æ–∑–¥–∞–Ω–∏–µ —Ç—Ä–µ—É–≥–æ–ª—å–Ω–æ–≥–æ –ø–æ–ª—è –ø–ª–∏–Ω–∫–æ –∫–∞–∫ –≤ 1win
        function createPlinkoField() {
            const board = document.getElementById('plinko-board');
            board.innerHTML = '';
            
            const rows = 12; // –ö–æ–ª–∏—á–µ—Å—Ç–≤–æ —Ä—è–¥–æ–≤ —à—Ç–∏—Ñ—Ç–æ–≤
            const pinRadius = 4; // –†–∞–¥–∏—É—Å —à—Ç–∏—Ñ—Ç–∞
            const rowHeight = 25; // –í—ã—Å–æ—Ç–∞ –º–µ–∂–¥—É —Ä—è–¥–∞–º–∏
            const boardWidth = board.offsetWidth;
            
            // –°–æ–∑–¥–∞–µ–º —à—Ç–∏—Ñ—Ç—ã –≤ —Ç—Ä–µ—É–≥–æ–ª—å–Ω–æ–º –ø–æ—Ä—è–¥–∫–µ
            for (let row = 0; row < rows; row++) {
                const pinsInRow = row + 1;
                const rowTop = 20 + row * rowHeight;
                
                // –†–∞–≤–Ω–æ–º–µ—Ä–Ω–æ–µ —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ —à—Ç–∏—Ñ—Ç–æ–≤ –ø–æ —à–∏—Ä–∏–Ω–µ
                const pinSpacing = boardWidth / (pinsInRow + 1);
                
                for (let col = 0; col < pinsInRow; col++) {
                    const pin = document.createElement('div');
                    pin.className = 'pin';
                    
                    // –ü–æ–∑–∏—Ü–∏–æ–Ω–∏—Ä–æ–≤–∞–Ω–∏–µ —à—Ç–∏—Ñ—Ç–æ–≤ –≤ —Ç—Ä–µ—É–≥–æ–ª—å–Ω–æ–º –ø–æ—Ä—è–¥–∫–µ
                    const left = pinSpacing * (col + 1);
                    
                    pin.style.left = left + 'px';
                    pin.style.top = rowTop + 'px';
                    pin.style.width = pinRadius * 2 + 'px';
                    pin.style.height = pinRadius * 2 + 'px';
                    
                    board.appendChild(pin);
                }
            }
        }

        // –ö–ò–î–ê–ï–ú –®–ê–†–ò–ö –ë–ï–ó –ó–ê–î–ï–†–ñ–ï–ö
        async function dropBall() {
            const betAmount = parseFloat(document.getElementById('bet-amount').value);
            if (betAmount < 0.1 || betAmount > 100) {
                showNotification('–°—Ç–∞–≤–∫–∞ –¥–æ–ª–∂–Ω–∞ –±—ã—Ç—å –æ—Ç 0.1 –¥–æ 100 TON', 'error');
                return;
            }

            const currentBalance = parseFloat(document.getElementById('current-balance').textContent);
            if (currentBalance < betAmount) {
                showNotification('–ù–µ–¥–æ—Å—Ç–∞—Ç–æ—á–Ω–æ —Å—Ä–µ–¥—Å—Ç–≤', 'error');
                return;
            }

            // –°–†–ê–ó–£ —Å–ø–∏—Å—ã–≤–∞–µ–º —Å—Ç–∞–≤–∫—É
            updateBalanceDisplay(currentBalance - betAmount);

            try {
                const response = await fetch('/api/plinko/drop', {
                    method: 'POST',
                    headers: {'Content-Type': 'application/json'},
                    body: JSON.stringify({
                        telegramId: currentUser.id,
                        betAmount: betAmount,
                        demoMode: isDemoMode
                    })
                });

                const result = await response.json();

                if (result.success) {
                    // –ó–∞–ø—É—Å–∫–∞–µ–º –∞–Ω–∏–º–∞—Ü–∏—é, –Ω–æ –ù–ï –æ–±–Ω–æ–≤–ª—è–µ–º –±–∞–ª–∞–Ω—Å –∑–¥–µ—Å—å
                    realisticPlinkoAnimation(result.slot_index, result.multiplier, result.win_amount, betAmount, result.new_balance);
                } else {
                    showNotification('–û—à–∏–±–∫–∞: ' + result.error, 'error');
                    // –í–æ–∑–≤—Ä–∞—â–∞–µ–º —Å—Ç–∞–≤–∫—É –ø—Ä–∏ –æ—à–∏–±–∫–µ
                    updateBalanceDisplay(currentBalance);
                }
            } catch (error) {
                console.error('Drop error:', error);
                showNotification('–û—à–∏–±–∫–∞ —Å–æ–µ–¥–∏–Ω–µ–Ω–∏—è', 'error');
                // –í–æ–∑–≤—Ä–∞—â–∞–µ–º —Å—Ç–∞–≤–∫—É –ø—Ä–∏ –æ—à–∏–±–∫–µ
                updateBalanceDisplay(currentBalance);
            }
        }

        // –†–ï–ê–õ–ò–°–¢–ò–ß–ù–ê–Ø –ê–ù–ò–ú–ê–¶–ò–Ø –ü–ê–î–ï–ù–ò–Ø –®–ê–†–ò–ö–ê (–±–µ–∑ –±–ª–æ–∫–∏—Ä–æ–≤–∫–∏ –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å–∞)
        function realisticPlinkoAnimation(targetSlot, multiplier, winAmount, betAmount, newBalance) {
            activeBalls++;
            
            const board = document.getElementById('plinko-board');
            const ball = document.createElement('div');
            ball.className = 'ball';
            board.appendChild(ball);

            const multipliers = document.querySelectorAll('.multiplier');
            const targetMultiplier = multipliers[targetSlot];

            // –û–ø—Ä–µ–¥–µ–ª—è–µ–º —Ü–µ–ª–µ–≤—É—é –ø–æ–∑–∏—Ü–∏—é –¥–ª—è —à–∞—Ä–∏–∫–∞
            const multiplierRects = Array.from(multipliers).map(m => m.getBoundingClientRect());
            const boardRect = board.getBoundingClientRect();
            const targetX = (multiplierRects[targetSlot].left + multiplierRects[targetSlot].width/2 - boardRect.left);

            // –ù–∞—á–∞–ª—å–Ω–∞—è –ø–æ–∑–∏—Ü–∏—è (—Ü–µ–Ω—Ç—Ä —Å–≤–µ—Ä—Ö—É)
            let x = boardRect.width / 2;
            let y = 0;
            
            ball.style.left = x + 'px';
            ball.style.top = y + 'px';

            const pins = document.querySelectorAll('.pin');
            
            // –ü—Ä–µ–¥–≤–∞—Ä–∏—Ç–µ–ª—å–Ω–æ –≤—ã—á–∏—Å–ª—è–µ–º –ø–æ–∑–∏—Ü–∏–∏ —à—Ç–∏—Ñ—Ç–æ–≤
            const pinPositions = Array.from(pins).map(pin => {
                const rect = pin.getBoundingClientRect();
                const boardRect = board.getBoundingClientRect();
                return {
                    x: rect.left + rect.width/2 - boardRect.left,
                    y: rect.top + rect.height/2 - boardRect.top,
                    radius: rect.width / 2
                };
            });

            // –§–∏–∑–∏—á–µ—Å–∫–∏–µ –ø–∞—Ä–∞–º–µ—Ç—Ä—ã
            let velocityX = 0;
            let velocityY = 0;
            const gravity = 0.3;
            const friction = 0.99;
            const ballRadius = 10;
            const bounceFactor = 0.7;

            let lastCollisionTime = 0;
            const collisionCooldown = 100; // –º—Å –º–µ–∂–¥—É —Å—Ç–æ–ª–∫–Ω–æ–≤–µ–Ω–∏—è–º–∏
            
            function animate() {
                const currentTime = Date.now();
                
                // –ü—Ä–∏–º–µ–Ω—è–µ–º –≥—Ä–∞–≤–∏—Ç–∞—Ü–∏—é
                velocityY += gravity;
                
                // –ü—Ä–∏–º–µ–Ω—è–µ–º —Ç—Ä–µ–Ω–∏–µ
                velocityX *= friction;
                velocityY *= friction;
                
                // –û–±–Ω–æ–≤–ª—è–µ–º –ø–æ–∑–∏—Ü–∏—é
                x += velocityX;
                y += velocityY;
                
                // –û–≥—Ä–∞–Ω–∏—á–∏–≤–∞–µ–º –¥–≤–∏–∂–µ–Ω–∏–µ –ø–æ –≥–æ—Ä–∏–∑–æ–Ω—Ç–∞–ª–∏ (—Å—Ç–µ–Ω–∫–∏)
                if (x < ballRadius) {
                    x = ballRadius;
                    velocityX *= -bounceFactor;
                }
                if (x > boardRect.width - ballRadius) {
                    x = boardRect.width - ballRadius;
                    velocityX *= -bounceFactor;
                }
                
                ball.style.left = x + 'px';
                ball.style.top = y + 'px';

                // –ü—Ä–æ–≤–µ—Ä—è–µ–º —Å—Ç–æ–ª–∫–Ω–æ–≤–µ–Ω–∏—è —Å–æ —à—Ç–∏—Ñ—Ç–∞–º–∏
                pinPositions.forEach((pin, index) => {
                    const dx = pin.x - x;
                    const dy = pin.y - y;
                    const distance = Math.sqrt(dx * dx + dy * dy);
                    const minDistance = pin.radius + ballRadius;
                    
                    // –ï—Å–ª–∏ –µ—Å—Ç—å —Å—Ç–æ–ª–∫–Ω–æ–≤–µ–Ω–∏–µ –∏ –ø—Ä–æ—à–ª–æ –¥–æ—Å—Ç–∞—Ç–æ—á–Ω–æ –≤—Ä–µ–º–µ–Ω–∏ —Å –ø–æ—Å–ª–µ–¥–Ω–µ–≥–æ
                    if (distance < minDistance && (currentTime - lastCollisionTime) > collisionCooldown) {
                        lastCollisionTime = currentTime;
                        
                        // –ù–æ—Ä–º–∞–ª–∏–∑—É–µ–º –≤–µ–∫—Ç–æ—Ä –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è
                        const nx = dx / distance;
                        const ny = dy / distance;
                        
                        // –û—Ç—Å–∫–∞–∫–∏–≤–∞–Ω–∏–µ –æ—Ç —à—Ç–∏—Ñ—Ç–∞
                        const dotProduct = velocityX * nx + velocityY * ny;
                        velocityX -= 2 * dotProduct * nx * bounceFactor;
                        velocityY -= 2 * dotProduct * ny * bounceFactor;
                        
                        // –î–æ–±–∞–≤–ª—è–µ–º —Å–ª—É—á–∞–π–Ω–æ–µ –æ—Ç–∫–ª–æ–Ω–µ–Ω–∏–µ –¥–ª—è —Ä–µ–∞–ª–∏—Å—Ç–∏—á–Ω–æ—Å—Ç–∏
                        velocityX += (Math.random() - 0.5) * 2;
                        
                        // –í–∏–∑—É–∞–ª—å–Ω—ã–π —ç—Ñ—Ñ–µ–∫—Ç —Å—Ç–æ–ª–∫–Ω–æ–≤–µ–Ω–∏—è
                        pins[index].style.background = '#ff9500';
                        pins[index].style.transform = 'scale(1.2)';
                        setTimeout(() => {
                            pins[index].style.background = '#00c851';
                            pins[index].style.transform = 'scale(1)';
                        }, 100);
                    }
                });

                // –ö–æ–≥–¥–∞ —à–∞—Ä–∏–∫ –¥–æ—Å—Ç–∏–≥–∞–µ—Ç –Ω–∏–∑–∞ (–æ–±–ª–∞—Å—Ç–∏ –º–Ω–æ–∂–∏—Ç–µ–ª–µ–π)
                if (y >= boardRect.height - 50) {
                    // –ü–ª–∞–≤–Ω–æ–µ –¥–≤–∏–∂–µ–Ω–∏–µ –∫ —Ü–µ–ª–µ–≤–æ–º—É –º–Ω–æ–∂–∏—Ç–µ–ª—é
                    function finalAnimation() {
                        // –ü–ª–∞–≤–Ω–æ–µ –ø—Ä–∏–±–ª–∏–∂–µ–Ω–∏–µ –∫ —Ü–µ–ª–µ–≤–æ–π –ø–æ–∑–∏—Ü–∏–∏
                        const dx = targetX - x;
                        x += dx * 0.1;
                        y += 2;
                        
                        ball.style.left = x + 'px';
                        ball.style.top = y + 'px';
                        
                        // –ö–æ–≥–¥–∞ —à–∞—Ä–∏–∫ –¥–æ—Å—Ç–∏–≥ —Ü–µ–ª–µ–≤–æ–π –ø–æ–∑–∏—Ü–∏–∏
                        if (Math.abs(dx) < 5 && y >= boardRect.height - 20) {
                            // –ü–æ–¥—Å–≤–µ—á–∏–≤–∞–µ–º –º–Ω–æ–∂–∏—Ç–µ–ª—å
                            targetMultiplier.classList.add('highlight');
                            
                            setTimeout(() => {
                                ball.remove();
                                targetMultiplier.classList.remove('highlight');
                                activeBalls--;
                                
                                // –û–±–Ω–æ–≤–ª—è–µ–º –±–∞–ª–∞–Ω—Å —Å –≤—ã–∏–≥—Ä—ã—à–µ–º
                                updateBalanceDisplay(newBalance);
                                
                                // –ü–æ–∫–∞–∑—ã–≤–∞–µ–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç
                                showResult(multiplier, winAmount, betAmount);
                            }, 800);
                            return;
                        }
                        
                        requestAnimationFrame(finalAnimation);
                    }
                    
                    requestAnimationFrame(finalAnimation);
                    return;
                }
                
                requestAnimationFrame(animate);
            }
            
            // –ù–∞—á–∞–ª—å–Ω—ã–π —Ç–æ–ª—á–æ–∫ –¥–ª—è —à–∞—Ä–∏–∫–∞
            velocityY = 2;
            velocityX = (Math.random() - 0.5) * 3; // –°–ª—É—á–∞–π–Ω–æ–µ –Ω–∞—á–∞–ª—å–Ω–æ–µ –æ—Ç–∫–ª–æ–Ω–µ–Ω–∏–µ
            
            requestAnimationFrame(animate);
        }

        // –ü–æ–∫–∞–∑ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–∞ (–æ–¥–Ω–æ —É–≤–µ–¥–æ–º–ª–µ–Ω–∏–µ –Ω–∞ 3 —Å–µ–∫—É–Ω–¥—ã)
        function showResult(multiplier, winAmount, betAmount) {
            const message = multiplier >= 1 ? 
                `üéâ –í—ã–∏–≥—Ä—ã—à: ${winAmount.toFixed(2)} TON (x${multiplier})` : 
                `üí∏ –ü—Ä–æ–∏–≥—Ä—ã—à: -${betAmount.toFixed(2)} TON`;
            
            showNotification(message, multiplier >= 1 ? 'success' : 'error');
        }

        // –§—É–Ω–∫—Ü–∏—è –ø–æ–∫–∞–∑–∞ —É–≤–µ–¥–æ–º–ª–µ–Ω–∏—è
        function showNotification(message, type) {
            const notification = document.getElementById('result-notification');
            const messageElement = document.getElementById('notification-message');
            
            messageElement.textContent = message;
            notification.className = `notification ${type}`;
            notification.style.display = 'block';
            
            // –ê–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–æ–µ –∑–∞–∫—Ä—ã—Ç–∏–µ —á–µ—Ä–µ–∑ 3 —Å–µ–∫—É–Ω–¥—ã
            setTimeout(() => {
                notification.style.display = 'none';
            }, 3000);
        }

        // –§—É–Ω–∫—Ü–∏–∏ –ø–æ–ø–æ–ª–Ω–µ–Ω–∏—è –±–∞–ª–∞–Ω—Å–∞

        function closeDepositModal() {

        async function processDeposit() {
            const amount = parseFloat(document.getElementById('deposit-amount').value);
            if (amount < 1) {
                showNotification('–ú–∏–Ω–∏–º–∞–ª—å–Ω–æ–µ –ø–æ–ø–æ–ª–Ω–µ–Ω–∏–µ: 1 TON', 'error');
                return;
            }

            try {
                const response = await fetch('/api/create-invoice', {
                    method: 'POST',
                    headers: {'Content-Type': 'application/json'},
                    body: JSON.stringify({
                        telegramId: currentUser.id,
                        amount: amount,
                        demoMode: isDemoMode
                    })
                });

                const result = await response.json();
                if (result.success) {
                    window.open(result.pay_url, '_blank');
                    checkDepositStatus(result.invoice_id);
                    closeDepositModal();
                } else {
                    showNotification('–û—à–∏–±–∫–∞: ' + result.error, 'error');
                }
            } catch (error) {
                showNotification('–û—à–∏–±–∫–∞ —Å–æ–µ–¥–∏–Ω–µ–Ω–∏—è', 'error');
            }
        }

        async function checkDepositStatus(invoiceId) {
            const checkInterval = setInterval(async () => {
                try {
                    const response = await fetch('/api/check-invoice', {
                        method: 'POST',
                        headers: {'Content-Type': 'application/json'},
                        body: JSON.stringify({ invoiceId, demoMode: isDemoMode })
                    });
                    
                    const result = await response.json();
                    
                    if (result.status === 'paid') {
                        clearInterval(checkInterval);
                        showNotification('‚úÖ –î–µ–ø–æ–∑–∏—Ç –∑–∞—á–∏—Å–ª–µ–Ω!', 'success');
                        loadUserData();
                    } else if (result.status === 'expired') {
                        clearInterval(checkInterval);
                        showNotification('‚ùå –í—Ä–µ–º—è –æ–ø–ª–∞—Ç—ã –∏—Å—Ç–µ–∫–ª–æ', 'error');
                    }
                } catch (error) {
                    console.error('Status check error:', error);
                }
            }, 3000);
        }

            }
